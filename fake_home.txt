  //    or we open a new page? But since it's a single document, we show "Page: X" in feed.
  //    Requirement: "All this buttons links to pages" means each button leads to separate page.
  //    For user experience, we load different page content inside the main scrollable area,
  //    preserving the global header, footer, nav. Also window.location can be used?

  // active page state + remove active highlight
  function setActiveButton(activeId) {
    const btns = ['navHome', 'navMessages', 'navAdd', 'navSaved', 'navSettings'];
    btns.forEach(id => {
      const btn = document.getElementById(id);
      if (btn) {
        if (id === activeId) btn.classList.add('active');
        else btn.classList.remove('active');
      }
    });
  }

  // navigation handler: changes page content dynamically and updates active style.
  // Also simulates "links to pages" because each button leads to distinct "page view".
  function navigateTo(page) {
    switch(page) {
      case 'home':
        renderHomePage();
        setActiveButton('navHome');
        break;
      case 'messages':
        renderMessagesPage();
        setActiveButton('navMessages');
        break;
      case 'add':
        renderAddPage();
        setActiveButton('navAdd');
        break;
      case 'saved':
        renderSavedPage();
        setActiveButton('navSaved');
        break;
      case 'settings':
        renderSettingsPage();
        setActiveButton('navSettings');
        break;
      default:
        renderHomePage();
        setActiveButton('navHome');
    }
    // optional: scroll to top when page changes
    document.querySelector('.scrollable-content').scrollTop = 0;
  }

  // attach click event listeners to the 5 image buttons
  document.getElementById('navHome').addEventListener('click', (e) => {
    e.preventDefault();
    navigateTo('home');
  });
  document.getElementById('navMessages').addEventListener('click', (e) => {
    e.preventDefault();
    navigateTo('messages');
  });
  document.getElementById('navAdd').addEventListener('click', (e) => {
    e.preventDefault();
    navigateTo('add');
  });
  document.getElementById('navSaved').addEventListener('click', (e) => {
    e.preventDefault();
    navigateTo('saved');
  });
  document.getElementById('navSettings').addEventListener('click', (e) => {
    e.preventDefault();
    navigateTo('settings');
  });

  // Search icon inside searchbar: alert search query (demo functionality)
  const searchIcon = document.querySelector('.search-icon-btn');
  const searchInput = document.querySelector('.search-bar');
  function performSearch() {
    const query = searchInput.value.trim();
    if(query) {
      alert(`🔍 Searching for: "${query}"\n(integrate with your backend in real app)`);
    } else {
      alert('Please enter something to search');
    }
  }
  searchIcon.addEventListener('click', performSearch);
  searchInput.addEventListener('keypress', (e) => {
    if(e.key === 'Enter') performSearch();
  });

  // initialize with Home page (rich moving content feed)
  navigateTo('home');
